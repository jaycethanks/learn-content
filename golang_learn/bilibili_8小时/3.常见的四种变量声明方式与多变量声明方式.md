## 变量声明

```go
package mian

// 四种变量的声明方式
func main() {
	// 方法一: 声明一个变量
	var a int // 未初始化是， int类型默认值是0

	// 方法二: 声明并显式初始化
	var b int = 100

	// 方法三: 在初始化的时候， 可以省去数据类型， 通过自动推导当前变量的数据类型
	var c = 100

	// 方法四: （最常用的方式）省区 var 关键字， 直接自动匹配
	// ! 但是 := 只能用于函数体中， 全局变量不可以这样声明
	e := 100

	// 声明多个变量
	// 同类型
	var xx, yy int = 100, 200
	// var nn,mm int = 100, "abcd" // !错误
	var kk, ll = 100, "abcd" //省略类型
	gg,hh := 100, "abcd"
	
	// 多行多变量的声明
	var(
		vv int = 100
		jj bool = true
	)
	// 可省略类型
	var(
		oo = 100
		pp = true
	)
}

```
